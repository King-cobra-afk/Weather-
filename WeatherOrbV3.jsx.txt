function getTheme(desc, temp) {
  const d = (desc||"").toLowerCase();
  if (d.includes("thunder")) return "thunder";
  if (d.includes("snow") || d.includes("blizzard")) return "snow";
  if (d.includes("fog") || d.includes("mist") || d.includes("haze")) return "fog";
  if (d.includes("rain") || d.includes("drizzle") || d.includes("shower")) return "rain";
  if (d.includes("cloud") || d.includes("overcast")) return "cloudy";
  if (temp >= 38) return "hot";
  return "clear";
}

function GaugeSVG({ val, max, color, unit }) {
  const r=30, cx=40, cy=40, sw=6;
  const circ = 2*Math.PI*r;
  const dash = Math.min(1, val/max) * circ * 0.75;
  return (
    <svg width="80" height="80" viewBox="0 0 80 80">
      <circle cx={cx} cy={cy} r={r} fill="none" stroke="rgba(255,255,255,0.08)" strokeWidth={sw}
        strokeDasharray={`${circ*0.75} ${circ}`} transform={`rotate(-225 ${cx} ${cy})`} strokeLinecap="round"/>
      <circle cx={cx} cy={cy} r={r} fill="none" stroke={color} strokeWidth={sw}
        strokeDasharray={`${dash} ${circ}`} transform={`rotate(-225 ${cx} ${cy})`} strokeLinecap="round"
        style={{filter:`drop-shadow(0 0 4px ${color})`}}/>
      <text x={cx} y={cy-3} textAnchor="middle" fontFamily="monospace" fontSize="11" fill="white" fontWeight="700">{val}</text>
      <text x={cx} y={cy+10} textAnchor="middle" fontSize="7" fill="rgba(255,255,255,0.45)">{unit}</text>
    </svg>
  );
}

function CompassSVG({ deg, speed }) {
  const cx=40, cy=40, r=28;
  const rad = (deg-90)*Math.PI/180;
  const ax = cx+r*0.6*Math.cos(rad), ay = cy+r*0.6*Math.sin(rad);
  return (
    <svg width="80" height="80" viewBox="0 0 80 80">
      <circle cx={cx} cy={cy} r={r} fill="none" stroke="rgba(255,255,255,0.08)" strokeWidth="1.5"/>
      {["N","E","S","W"].map((d,i)=>{
        const a=(i*90-90)*Math.PI/180;
        return <text key={d} x={cx+r*Math.cos(a)} y={cy+r*Math.sin(a)+3} textAnchor="middle" fontSize="7" fill="rgba(255,255,255,0.4)" fontFamily="monospace">{d}</text>;
      })}
      <line x1={cx} y1={cy} x2={ax} y2={ay} stroke="#ffd54f" strokeWidth="2.5" strokeLinecap="round" style={{filter:"drop-shadow(0 0 4px #ffd54f)"}}/>
      <circle cx={cx} cy={cy} r="3" fill="#ffd54f"/>
      <text x={cx} y={cy+17} textAnchor="middle" fontFamily="monospace" fontSize="9" fill="white" fontWeight="700">{speed}</text>
      <text x={cx} y={cy+25} textAnchor="middle" fontSize="6" fill="rgba(255,255,255,0.45)">KM/H</text>
    </svg>
  );
}

function RainDrops() {
  const drops = Array.from({length:35},(_,i)=>({
    l:Math.floor(Math.random()*100),
    h:Math.floor(Math.random()*40+20),
    o:(Math.random()*0.5+0.2).toFixed(2),
    d:(Math.random()*0.5+0.6).toFixed(2),
    dl:(Math.random()*2).toFixed(2)
  }));
  return (
    <div style={{position:"fixed",inset:0,pointerEvents:"none",zIndex:0,overflow:"hidden"}}>
      {drops.map((d,i)=>(
        <div key={i} style={{
          position:"absolute",top:"-20px",left:`${d.l}%`,width:"1.5px",height:`${d.h}px`,
          background:"linear-gradient(transparent,rgba(79,195,247,0.6))",
          opacity:d.o,animation:`rainFall ${d.d}s linear ${d.dl}s infinite`
        }}/>
      ))}
    </div>
  );
}

async function callClaude(locationStr) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: `You are a weather data provider. Given a location (city name or lat/lon coordinates), respond ONLY with a valid JSON object — no markdown, no explanation, no extra text. Use realistic weather data for that location based on current season and typical climate. JSON structure:
{"city":"City Name","country":"Country","temp":32,"feels":35,"desc":"Partly Cloudy","icon_code":2,"humidity":65,"wind_speed":14,"wind_dir":180,"uv":7.2,"pressure":1008,"rain":0.0,"visibility":10,"sunrise":"06:12","hourly":[{"hour":"NOW","temp":32,"rain_pct":10,"icon_code":2},{"hour":"1PM","temp":33,"rain_pct":5,"icon_code":1},{"hour":"2PM","temp":34,"rain_pct":0,"icon_code":0},{"hour":"3PM","temp":33,"rain_pct":15,"icon_code":2},{"hour":"4PM","temp":31,"rain_pct":30,"icon_code":61},{"hour":"5PM","temp":29,"rain_pct":45,"icon_code":63},{"hour":"6PM","temp":28,"rain_pct":20,"icon_code":80},{"hour":"7PM","temp":27,"rain_pct":10,"icon_code":2},{"hour":"8PM","temp":26,"rain_pct":5,"icon_code":1}],"weekly":[{"day":"TODAY","high":34,"low":24,"icon_code":2},{"day":"TUE","high":35,"low":25,"icon_code":0},{"day":"WED","high":32,"low":23,"icon_code":61},{"day":"THU","high":30,"low":22,"icon_code":63},{"day":"FRI","high":33,"low":24,"icon_code":2},{"day":"SAT","high":36,"low":26,"icon_code":0},{"day":"SUN","high":34,"low":25,"icon_code":1}]}`,
      messages: [{ role: "user", content: `Weather for: ${locationStr}` }]
    })
  });
  const data = await res.json();
  const txt = data.content?.find(b => b.type === "text")?.text || "";
  return JSON.parse(txt.replace(/```json|```/g, "").trim());
}

export default function WeatherOrb() {

  const [theme, setTheme] = useState("clear");

  const t = THEMES[theme];
  const gls = "rgba(255,255,255,0.07)";
  const glsB = "rgba(255,255,255,0.15)";
  const isSnow = theme === "snow";
  const textColor = isSnow ? "#0a1628" : "#e8f4fd";
  const dimColor = isSnow ? "rgba(10,22,40,0.5)" : "rgba(232,244,253,0.5)";

  useEffect(() => { requestLocation(); }, []);

    setStatus("locating");

      async (pos) => {
        setStatus("loading");
        try {
          const { latitude: lat, longitude: lon } = pos.coords;
          // Reverse geocode with nominatim
          let locationStr = `${lat.toFixed(4)}, ${lon.toFixed(4)}`;
          try {
            const r = await fetch(`https://nominatim.openstreetmap.org/reverse?lat=${lat}&lon=${lon}&format=json`);
            const rd = await r.json();
            const city = rd.address.city || rd.address.town || rd.address.village || rd.address.county || "";
            const country = rd.address.country || "";
            if (city) locationStr = `${city}, ${country}`;
          } catch {}
          const w = await callClaude(locationStr);
          setTheme(getTheme(w.desc, w.temp));
          setWeather(w);
          setStatus("done");
        } catch(e) {
          setStatus("denied");
        }
      },
      () => setStatus("denied"),
      { timeout: 10000, maximumAge: 300000 }
    );
  }

    background: gls, border: `1px solid ${glsB}`,

        @keyframes rainFall{to{transform:translateY(110vh)}}

      {(theme==="rain"||theme==="thunder") && <RainDrops/>}

      <div style={{position:"relative",zIndex:1,maxWidth:"430px",margin:"0 auto",padding:"24px 16px 40px"}}>

        {/* LOCATING */}
        {status==="locating" && (
          <div style={{textAlign:"center",paddingTop:"120px"}}>

            <div style={{fontFamily:"Orbitron,monospace",fontSize:"14px",letterSpacing:"3px",color:"rgba(255,255,255,0.6)",textTransform:"uppercase",animation:"blink 1.5s infinite"}}>

            </div>
            <div style={{fontSize:"11px",color:"rgba(255,255,255,0.35)",marginTop:"10px",letterSpacing:"2px"}}>
              Allow karr popup mein 👆

            </div>
          </div>
